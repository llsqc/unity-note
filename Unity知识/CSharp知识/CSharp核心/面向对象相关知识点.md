### 命名空间
```c#
//public   —— 命名空间中的类 
//internal —— 只能在该程序集中使用，默认为internal
//abstract —— 抽象类
//sealed   —— 密封类
//partial  —— 分部类

//总结
//1.命名空间是个工具包 用来管理类的
//2.不同命名空间中 可以有同名类
//3.不同命名空间中相互使用 需要using引用命名空间 或者 指明出处
//4.命名空间可以包裹命名空间
```

### String
```c#
// 字符串拼接
str = string.Format("{0}{1}", 1, 3333);

/// 正向查找字符位置
str = "我是唐老狮！";
int index = str.IndexOf("唐");
index = str.IndexOf("师"); // -1

// 反向查找指定字符串位置
str = "我是唐老狮唐老狮";
index = str.LastIndexOf("唐老狮");
index = str.LastIndexOf("唐老师"); // -1

// 移除指定位置后的字符
str = "我是唐老狮唐老狮";
str = str.Remove(4);

//执行两个参数进行移除
//参数1 开始位置 参数2 字符个数
str = str.Remove(1, 1);

// 替换指定字符串
str = "我是唐老狮唐老狮";
str = str.Replace("唐老狮", "老炮儿");

// 大小写转换
str = "ksdfasdfasfasdfsasdfasdf";
str = str.ToUpper();
str = str.ToLower();

// 字符串截取
str = "唐老狮唐老狮";
//截取从指定位置开始之后的字符串
str = str.Substring(2);

//参数一 开始位置 参数二 指定个数
str = str.Substring(2, 2);

// 字符串切割
str = "1_1|2_2|3_3|5_1|6_1|7_2|8_3";
string[] strs = str.Split('|');
```

### StringBuilder
```c#
using System.Text;

// 直接指明内容
StringBuilder str = new StringBuilder("123123123");
//StringBuilder存在一个容量的问题，每次往里面增加时 会自动扩容
//获得容量
Console.WriteLine(str.Capacity);
//获得字符长度
Console.WriteLine(str.Length);
//增
str.Append("4444");
str.AppendFormat("{0}{1}", 100, 999);
//插入
str.Insert(0, "唐老狮");
//删
str.Remove(0, 10);
//清空
str.Clear();
//查
Console.WriteLine(str[1]);
//改
str[0] = 'A';
//替换
str.Replace("1", "唐");
//重新赋值 StringBuilder
str.Clear();
str.Append("123123");
//判断StringBuilder是否和某一个字符串相等
if (str.Equals("123123"))
{
    Console.WriteLine("相等");
}
```
