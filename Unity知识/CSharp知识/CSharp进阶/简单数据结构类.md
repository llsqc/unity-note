### ArrayList
```c#
using System.Collections;

// 添加元素
array.Add(1);
array.Add("123");
array.Add(true);

// 指定位置插入元素
array.Insert(1, "12345676");

//移除指定元素 从头找 找到删
array.Remove(1);

//移除指定位置的元素
array.RemoveAt(2);

//清空
array.Clear();

//得到指定位置的元素
Console.WriteLine(array[0]);

//查看元素是否存在
bool bExist = array.Contains("1234");

//正向查找元素位置，找到的返回值 是位置 找不到 返回值 是-1
int index = array.IndexOf(true);

//反向查找元素位置，返回时从头开始的索引数
index = array.LastIndexOf(true);

// 修改指定位置的元素
array[0] = "999";

//长度
Console.WriteLine(array.Count);

//容量 (避免产生过多的垃圾)
Console.WriteLine(array.Capacity);

//迭代器
foreach (object item in array)
{
    Console.WriteLine(item);
}

//存在装箱拆箱操作
int k = 1;
array[0] = k;//装箱
k = (int)array[0];//拆箱
```

### Stack
```c#
using System.Collections;

Stack stack = new Stack();

//压栈
stack.Push(1);
stack.Push("123");

//弹栈
object v = stack.Pop();

//查看栈顶的内容
v = stack.Peek();

//查看元素是否存在于栈中
bool bExist = stack.Contains("123");

//清空栈
stack.Clear();

//长度
Console.WriteLine(stack.Count);

//foreach遍历 从栈顶到栈底
foreach (object item in stack)
{
    Console.WriteLine(item);
}

//  将栈转换为object数组 遍历出来的顺序 从栈顶到栈底
object[] array = stack.ToArray();
```

### Queue
```c#
using System.Collections;

Queue queue = new Queue();

//入队
queue.Enqueue(1);
queue.Enqueue("123");
queue.Enqueue(1.4f);

//出队
object v = queue.Dequeue();

//查看队列头部元素但不会移除
v = queue.Peek();


//查看元素是否存在于队列中
bool bExist = queue.Contains(1.4f);

//清空队列
queue.Clear();

//长度
Console.WriteLine(queue.Count);

//用foreach遍历
foreach (object item in queue)
{
    Console.WriteLine(item);
}

//将队列转换为object数组
object[] array = queue.ToArray();
```

### Hashtable
```c#
using System.Collections;

Hashtable hashtable = new Hashtable();

//增（注意：不能出现相同键）
hashtable.Add(1, "123");
hashtable.Add("123", 2);
hashtable.Add(true, false);

//只能通过键去删除 删除不存在的键 没反应
hashtable.Remove(1);

//直接清空
hashtable.Clear();

//通过键查看值，找不到会返回空
Console.WriteLine(hashtable["123123"]);

//查看是否存在
//根据键检测
if (hashtable.Contains(2))
{
    Console.WriteLine("存在键为2的键值对");
}
//根据值检测
if (hashtable.ContainsValue(12))
{
    Console.WriteLine("存在值为12的键值对");
}

//只能改 键对应的值内容 无法修改键
hashtable[1] = 100.5f;

//得到键值对的对数
Console.WriteLine(hashtable.Count);

//遍历所有键
foreach (object item in hashtable.Keys)
{
    Console.WriteLine("键：" + item);
    Console.WriteLine("值：" + hashtable[item]);
}

//遍历所有值
foreach (object item in hashtable.Values)
{
    Console.WriteLine("值：" + item);
}

//键值对一起遍历
foreach (DictionaryEntry item in hashtable)
{
    Console.WriteLine("键：" + item.Key + "值：" + item.Value);
}

//迭代器遍历法
IDictionaryEnumerator myEnumerator = hashtable.GetEnumerator();
bool flag = myEnumerator.MoveNext();
while (flag)
{
    Console.WriteLine("键：" + myEnumerator.Key + "值：" + myEnumerator.Value);
    flag = myEnumerator.MoveNext();
}
```
